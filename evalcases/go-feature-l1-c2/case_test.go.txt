package go_feature_l1_c2

import (
	"testing"

	"github.com/ZacharyFan/ai-coding-evaluation-demo-golang/pricing"
)

func TestFormatCents(t *testing.T) {
	tests := map[int]string{
		12345: "USD 123.45",
		0:     "USD 0.00",
		-123:  "-USD 1.23",
	}
	for cents, want := range tests {
		if got := pricing.FormatCents(cents, "USD"); got != want {
			t.Fatalf("FormatCents(%d, USD) = %q, want %q", cents, got, want)
		}
	}
}
