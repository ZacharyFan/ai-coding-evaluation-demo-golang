package go_bugfix_l1_c2

import (
	"testing"

	"github.com/ZacharyFan/ai-coding-evaluation-demo-golang/pricing"
)

func TestCalculateTaxRoundsHalfUpToCent(t *testing.T) {
	got, err := pricing.CalculateTax(200, 825)
	if err != nil {
		t.Fatalf("CalculateTax returned error: %v", err)
	}
	if got != 17 {
		t.Fatalf("CalculateTax(200, 825) = %d, want 17", got)
	}
}
