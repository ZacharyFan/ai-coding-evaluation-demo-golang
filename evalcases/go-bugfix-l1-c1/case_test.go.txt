package go_bugfix_l1_c1

import (
	"testing"
	"github.com/ZacharyFan/ai-coding-evaluation-demo-golang/catalog"
)

func TestValidateProductRejectsBlankSKU(t *testing.T) {
	err := catalog.ValidateProduct(catalog.Product{SKU: "  ", Name: "Notebook", PriceCents: 1200, Status: catalog.StatusActive})
	if err == nil {
		t.Fatal("ValidateProduct accepted blank SKU; want error")
	}
}
