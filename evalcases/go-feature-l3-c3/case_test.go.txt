package go_feature_l3_c3

import (
	"reflect"
	"testing"
	"github.com/ZacharyFan/ai-coding-evaluation-demo-golang/settlement_reconciliation"
)

func TestBuildReportSortsDiscrepancies(t *testing.T) {
	report := settlement_reconciliation.BuildReport(
		[]settlement_reconciliation.Payout{{ExternalID: "b", AmountCents: 100}, {ExternalID: "a", AmountCents: 100}},
		[]settlement_reconciliation.LedgerEntry{{ExternalID: "b", AmountCents: 90}},
	)
	got := []string{}
	for _, d := range report.Discrepancies { got = append(got, d.ExternalID+":"+d.Reason) }
	want := []string{"a:missing ledger", "b:amount mismatch"}
	if !reflect.DeepEqual(got, want) {
		t.Fatalf("BuildReport discrepancies = %#v, want %#v", got, want)
	}
}
