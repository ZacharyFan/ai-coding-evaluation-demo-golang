package go_bugfix_l3_c2

import (
	"testing"

	"github.com/ZacharyFan/ai-coding-evaluation-demo-golang/inventory"
)

func TestReserveManyIsAtomicOnFailure(t *testing.T) {
	ledger := inventory.NewLedger(map[string]int{"A": 5, "B": 1})
	err := ledger.ReserveMany(map[string]int{"A": 3, "B": 2})
	if err == nil {
		t.Fatal("ReserveMany returned nil error; want insufficient stock error")
	}
	if got := ledger.Snapshot()["A"]; got != 5 {
		t.Fatalf("A availability after failed batch = %d, want 5", got)
	}
}
